cmake_minimum_required (VERSION 3.6)
set(CMAKE_BUILD_TYPE Release)
set(REPO_ROOT ${CMAKE_CURRENT_LIST_DIR}/..)
set (CMAKE_INSTALL_PREFIX ${REPO_ROOT}/artifacts)

if (WIN32)
    add_subdirectory(win7-x64)
elseif(UNIX)
    if (APPLE)
        add_subdirectory(osx-x64)
    else()
        add_subdirectory(linux-x64)
        add_subdirectory(linux-arm)
        add_subdirectory(linux-arm64)
        add_subdirectory(linux-armel)
    endif()
else()
     message( FATAL_ERROR "Could not determine OS")
endif()
